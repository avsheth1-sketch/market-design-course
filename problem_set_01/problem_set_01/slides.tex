\begin{frame}{Market Design}
Design rules and algorithms to improve market outcomes.

\begin{frame}{Why Markets Fail}
\begin{itemize}
\item Congestion
\item Incentives problems
\item Information asymmetry
\end{itemize}
\end{frame}

\begin{frame}{Diagnosing Market Failures}
\begin{table}
\small
\begin{tabular}{p{3cm} p{6cm}}
\textbf{Failure Type} & \textbf{Example} \\
\hline
Externalities & Pollution in cap-and-trade markets where firms do not internalize social costs \\
Information Asymmetry & Adverse selection in insurance markets (e.g. health insurance) \\
Market Power & Monopoly pricing in electricity or telecom markets \\
Coordination Failure & Matching markets failing without centralized mechanisms (e.g. kidney exchange) \\
\end{tabular}
\end{table}
\end{frame}

\begin{frame}{Design Tools}
\begin{itemize}
\item Auctions
\item Matching
\item Pricing rules
\end{itemize}
\end{frame}

\begin{frame}{Matching Mechanisms}
Deferred Acceptance (stable matching).
\end{frame}

\begin{frame}{Evaluating and Comparing Designs}
\begin{itemize}
\item \textbf{Efficiency:} Maximizes total welfare and minimizes wasted resources
\item \textbf{Stability:} No participants prefer to deviate or form alternative matches
\item \textbf{Incentives:} Truthful reporting is optimal (strategy-proofness)
\item \textbf{Fairness:} Respects priorities and ensures equitable outcomes
\item \textbf{Practicality:} Simple, transparent, and feasible to implement
\end{itemize}
\end{frame}


\begin{frame}{Applications}
School choice, residency matching, housing, organs.
\end{frame}

\begin{frame}{Tradeoffs}
Fairness vs efficiency, simplicity vs optimality.
\end{frame}

\begin{frame}{Takeaways}
Good design improves welfare and allocation.
\end{frame}
